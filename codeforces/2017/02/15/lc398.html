from random import randint

class Solution(object):

    def __init__(self, nums):
        """
        
        :type nums: List[int]
        :type numsSize: int
        """
        self.nums = nums
        

    def pick(self, target):
        """
        :type target: int
        :rtype: int
        """
        ret = -1
        count = 0
        for i, x in enumerate(self.nums):
            if x == target:
                if randint(0, count) == 0:
                    ret = i
                count += 1
        return ret
