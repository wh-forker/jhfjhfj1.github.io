#include <cstdio>
#include <algorithm>
using namespace std;

const int MAX_N = 110;
const int MAX_M = 110;

int n, m, opr_num;
int f[MAX_N][MAX_M];
int vec[MAX_N];

void make(int bits)
{
	int i = 0;
	while (bits)
	{
		vec[i++] = bits & 1;
		bits >>= 1;
	}
}

int work()
{
	int ret = 0;
	for (int i = 0; i < m; i++)
	{
		int same = 0;
		int diff = 0;
		for (int j = 0; j < n; j++)
		{
			if (vec[j] == f[j][i])
			{
				same++;
			}else
			{
				diff++;
			}
		}
		ret += min(same, diff);
	}
	return ret;
}

int main()
{
	//input
	scanf("%d%d%d", &n, &m, &opr_num);
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			if (m > n)
			{
				scanf("%d", &f[i][j]);
			}else
			{
				scanf("%d", &f[j][i]);
			}
		}
	}
	if (n > m)
	{
		swap(m, n);
	}

	//work
	int ans = n * m;
	if (n <= opr_num)
	{
		for (int i = 0; i < (1 << n); i++)
		{
			make(i);
			ans = min(ans, work());
		}
	}else
	{
		for (int i = 0; i < m; i++)
		{
			for (int j = 0; j < n; j++)
			{
				vec[j] = f[j][i];
			}
			ans = min(ans, work());
		}
	}
	if (ans > opr_num)
	{
		puts("-1");
	}else
	{
		printf("%d\n", ans);
	}
	return 0;
}
